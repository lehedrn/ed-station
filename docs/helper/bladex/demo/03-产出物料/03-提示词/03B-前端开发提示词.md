## 使用说明

本提示词文档用于指导完成前端开发阶段的任务：
1. **API 接口封装** - 统一的接口请求封装

4. **详情页面组件** - 详情展示页面

```javascript
/**
 * {模块名称} API 接口封装
 * @author {作者}
 * @date {日期}
 */

import request from '@/axios';

/**
 * 分页查询{模块名称}列表
 * @param {number} current - 当前页码
 * @param {number} size - 每页条数
 * @param {object} params - 查询参数
 * @param {string} params.affairName - 事项名称（模糊查询）
 * @param {string} params.affairType - 事项类别（精确匹配）
 * @param {number} params.status - 事项状态
 * @returns {Promise}
 */
export const getList = (current, size, params) => {
  return request({
    url: '/blade-{模块目录}/{模块代码}/list',
    method: 'get',
    params: { ...params, current, size },
    cryptoToken: false,
    cryptoData: false,
  });
};

/**
 * 查询{模块名称}详情
 * @param {string|number} id - 事项 ID
 * @returns {Promise}
 */
export const getDetail = (id) => {
  return request({
    url: '/blade-{模块目录}/{模块代码}/detail',
    method: 'get',
    params: { id },
    cryptoToken: false,
    cryptoData: false,
  });
};

/**
 * 新增{模块名称}
 * @param {object} data - 事项数据
 * @returns {Promise}
 */
export const save = (data) => {
  return request({
    url: '/blade-{模块目录}/{模块代码}/save',
    method: 'post',
    data,
    cryptoToken: false,
    cryptoData: false,
  });
};

/**
 * 修改{模块名称}
 * @param {object} data - 事项数据（包含 id）
 * @returns {Promise}
 */
export const update = (data) => {
  return request({
    url: '/blade-{模块目录}/{模块代码}/update',
    method: 'post',
    data,
    cryptoToken: false,
    cryptoData: false,
  });
};

/**
 * 删除{模块名称}（支持批量）
 * @param {string} ids - 事项 ID 列表（逗号分隔）
 * @returns {Promise}
 */
export const remove = (ids) => {
  return request({
    url: '/blade-{模块目录}/{模块代码}/remove',
    method: 'post',
    data: { ids },
    cryptoToken: false,
    cryptoData: false,
  });
};

/**
 * 发布{模块名称}
 * @param {string|number} id - 事项 ID
 * @returns {Promise}
 */
export const publish = (id) => {
  return request({
    url: '/blade-{模块目录}/{模块代码}/publish',
    method: 'post',
    data: { id },
    cryptoToken: false,
    cryptoData: false,
  });
};

/**
 * 下架{模块名称}
 * @param {string|number} id - 事项 ID
 * @returns {Promise}
 */
export const unpublish = (id) => {
  return request({
    url: '/blade-{模块目录}/{模块代码}/unpublish',
    method: 'post',
    data: { id },
    cryptoToken: false,
    cryptoData: false,
  });
};

/**
 * 上传文件
 * @param {File} file - 文件对象
 * @returns {Promise}
 */
export const uploadFile = (file) => {
  const formData = new FormData();
  formData.append('file', file);
  return request({
    url: '/blade-resource/attach/upload',
    method: 'post',
    data: formData,
    headers: {
      'Content-Type': 'multipart/form-data',
    },
    cryptoToken: false,
    cryptoData: false,
  });
};

/**
 * 下载文件
 * @param {string|number} id - 附件 ID
 * @returns {string}
 */
export const downloadUrl = (id) => {
  return `/blade-resource/attach/download?id=${id}`;
};
```

          :accept="'.pdf,.doc,.docx,.jpg,.png'"
          :limit="1"
        >
          <el-button type="primary">选择文件</el-button>
          <template #tip>
            <div class="el-upload__tip">
              支持 pdf/doc/docx/jpg/png 格式，单文件≤20MB
            </div>
          </template>
        </el-upload>
      </el-form-item>
    </el-form>
    
    <template #footer>
      <el-button @click="handleClose">取消</el-button>
      <el-button type="primary" @click="handleSubmit">确认</el-button>
    </template>
  </el-dialog>

<script>
import { uploadFile } from '@/api/{模块目录}/{模块代码}';

export default {
  name: 'MaterialForm',
  props: {
    modelValue: {
      type: Boolean,
      default: false
    },
    data: {
      type: Object,
      default: () => ({})
    }
  },
  emits: ['update:modelValue', 'submit'],
  data() {
    return {
      dialogVisible: false,
      isEdit: false,
      form: {
        materialName: '',
        materialType: '',
        materialCopies: 1,
        materialRemark: '',
        attachId: null
      },
      fileList: [],
      uploadUrl: '/blade-resource/attach/upload',
      materialTypeOptions: [], // 从字典加载
      rules: {
        materialName: [
          { required: true, message: '请输入材料名称', trigger: 'blur' }
        ],
        materialType: [
          { required: true, message: '请选择材料类型', trigger: 'change' }
        ],
        attachId: [
          { required: true, message: '请上传附件文件', trigger: 'change' }
        ]
      }
    };
  },
  watch: {
    modelValue(val) {
      this.dialogVisible = val;
    },
    dialogVisible(val) {
      this.$emit('update:modelValue', val);
      if (val) {
        this.loadDict();
        if (this.data && this.data.id) {
          this.isEdit = true;
          this.form = { ...this.data };
        } else {
          this.isEdit = false;
          this.resetForm();
        }
      }
    }
  },
  methods: {
    // 加载字典
    async loadDict() {
      // 从字典接口加载材料类型
    },
    
    // 重置表单
    resetForm() {
      this.form = {
        materialName: '',
        materialType: '',
        materialCopies: 1,
        materialRemark: '',
        attachId: null
      };
      this.fileList = [];
    },
    
    // 文件上传前校验
    handleBeforeUpload(file) {
      const isValidType = ['application/pdf', 'application/msword', 
        'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
        'image/jpeg', 'image/png'].includes(file.type);
      const isValidSize = file.size <= 20 * 1024 * 1024;
      
      if (!isValidType) {
        this.$message.error('仅支持 pdf/doc/docx/jpg/png 格式');
        return false;
      }
      if (!isValidSize) {
        this.$message.error('单文件大小不能超过 20MB');
        return false;
      }
      return true;
    },
    
    // 文件上传成功
    handleSuccess(response) {
      if (response.code === 200) {
        this.form.attachId = response.data.id;
        this.fileList = [{
          name: response.data.fileName,
          url: response.data.filePath
        }];
      }
    },
    
    // 提交表单
    handleSubmit() {
      this.$refs.formRef.validate(async (valid) => {
        if (valid) {
          this.$emit('submit', { ...this.form, isEdit: this.isEdit });
          this.dialogVisible = false;
        }
      });
    },
    
    // 关闭弹窗
    handleClose() {
      this.dialogVisible = false;
    }
  }
};
</script>

import { getDetail } from '@/api/{模块目录}/{模块代码}';
import dayjs from 'dayjs';

export default {
  name: '{模块类名}Detail',
  data() {
    return {
      detail: {},
      previewVisible: false,
      previewFile: null
    };
  },
  computed: {
    isImage() {
      if (!this.previewFile) return false;
      return this.previewFile.fileType?.includes('image');
    },
    isPdf() {
      if (!this.previewFile) return false;
      return this.previewFile.fileType === 'application/pdf';
    }
  },
  async created() {
    await this.loadDetail();
  },
  methods: {
    async loadDetail() {
      try {
        const id = this.$route.query.id;
        const res = await getDetail(id);
        this.detail = res.data;
      } catch (error) {
        this.$message.error('加载详情失败：' + error.message);
      }
    },
    
    handleBack() {
      this.$router.back();
    },
    
    getStatusType(status) {
      const map = { 1: 'primary', 2: 'info', 3: 'danger' };
      return map[status] || 'info';
    },
    
    formatTime(time) {
      return time ? dayjs(time).format('YYYY-MM-DD HH:mm:ss') : '-';
    },
    
    handlePreview(attach) {
      this.previewFile = attach;
      this.previewVisible = true;
    },
    
    handleDownload(attach) {
      const link = document.createElement('a');
      link.href = `/blade-resource/attach/download?id=${attach.id}`;
      link.download = attach.fileName;
      link.click();
    },
    
    handleDownloadFile() {
      this.handleDownload(this.previewFile);
    }
  }
};
</script>
