## 2. API 接口设计

### 2.1 接口规范

#### 2.1.1 接口基础路径

```
/blade-[模块]/[资源]/[操作]
```

#### 2.1.2 请求/响应格式

**请求头**：

```
Content-Type: application/json
Authorization: Bearer {token}
Tenant-Id: {tenantId}
```

**统一响应格式（BladeX R<T>）**：

```json
{
  "code": 200,           // 状态码：200 成功，其他失败
  "success": true,       // 是否成功
  "data": {},            // 业务数据
  "msg": "操作成功"      // 提示信息
}
```

**分页响应格式**：

```json
{
  "code": 200,
  "success": true,
  "data": {
    "records": [],       // 数据列表
    "total": 100,        // 总记录数
    "current": 1,        // 当前页
    "size": 10,          // 每页条数
    "pages": 10          // 总页数
  },
  "msg": "查询成功"
}
```

#### 2.1.3 查询条件后缀（BladeX 规范）

| 后缀 | 说明 | 示例 |
|------|------|------|
| `_equal` | 等于 | `status_equal=1` |
| `_like` | 模糊 | `name_like=张` |
| `_ge` / `_le` | 大于/小于等于 | `age_ge=18` |
| `_gt` / `_lt` | 大于/小于 | `age_gt=18` |
| `_datege` / `_datelt` | 日期大于/小于等于 | `createTime_datege=2026-01-01` |

// src/api/[模块]/[资源].js
import request from '@/axios';

/**
 * 分页查询

  return request({
    url: '/blade-[模块]/[资源]/list',
    method: 'get',
    params: { ...params, current, size },
    cryptoToken: false,
    cryptoData: false,
  });
};

  return request({

    method: 'get',
    params: { id },
  });
};

/**
 * 新增
 * @param {object} row - 数据对象

  return request({
    url: '/blade-[模块]/[资源]/save',

    data: row,
  });

 * 修改

  return request({
    url: '/blade-[模块]/[资源]/update',
